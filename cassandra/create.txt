CREATE KEYSPACE deadman_event_log
  WITH replication = {'class': 'SimpleStrategy', 'replication_factor': '1'}
  AND durable_writes = true;

CREATE KEYSPACE deadman
  WITH replication = {'class': 'SimpleStrategy', 'replication_factor': '1'}
  AND durable_writes = true;

USE deadman;

CREATE TABLE aggregate_warning (
  aggregate TEXT,
  key TEXT,
  ttw BIGINT,
  entity TEXT,
  ttl BIGINT,
  creation BIGINT,
  warning BIGINT,
  tags TEXT,
  PRIMARY KEY (aggregate, key, ttw)
);

CREATE TABLE entity_warning (
  entity TEXT,
  key TEXT,
  ttw BIGINT,
  aggregate TEXT,
  ttl BIGINT,
  creation BIGINT,
  warning BIGINT,
  tags TEXT,
  PRIMARY KEY (entity, key, ttw)
);

CREATE TABLE aggregate_expiration (
  aggregate TEXT,
  entity TEXT,
  key TEXT,
  ttl BIGINT,
  creation BIGINT,
  expiration BIGINT,
  tags TEXT,
  PRIMARY KEY (aggregate, key)
);

CREATE TABLE entity_expiration (
  entity TEXT,
  aggregate TEXT,
  key TEXT,
  ttl BIGINT,
  creation BIGINT,
  expiration BIGINT,
  tags TEXT,
  PRIMARY KEY (entity, key)
);

CREATE TABLE tag_expiration (
  tag TEXT,
  window TEXT,
  expiration BIGINT,
  aggregate TEXT,
  entity TEXT,
  key TEXT,
  ttl BIGINT,
  creation BIGINT,
  tags TEXT,
  PRIMARY KEY ((tag, window), expiration)
);

CREATE TABLE write_progress (
  id TEXT,
  sequence_nr BIGINT,
  PRIMARY KEY (id)
);
