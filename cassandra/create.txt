  AND durable_writes = true;

USE deadman;

CREATE TABLE aggregate_warning (
  aggregate TEXT,
  key TEXT,
  ttw BIGINT,
  entity TEXT,
  ttl BIGINT,
  creation BIGINT,
  warning BIGINT,
  tags SET<TEXT>,
  PRIMARY KEY (aggregate, entity, key, ttw)
);

CREATE TABLE entity_warning (
  entity TEXT,
  key TEXT,
  ttw BIGINT,
  aggregate TEXT,
  ttl BIGINT,
  creation BIGINT,
  warning BIGINT,
  tags SET<TEXT>,
  PRIMARY KEY (entity, aggregate, key, ttw)
);

CREATE TABLE aggregate_expiration (
  aggregate TEXT,
  entity TEXT,
  key TEXT,
  ttl BIGINT,
  creation BIGINT,
  expiration BIGINT,
  tags SET<TEXT>,
  PRIMARY KEY (aggregate, entity, key)
);

CREATE TABLE entity_expiration (
  entity TEXT,
  aggregate TEXT,
  key TEXT,
  ttl BIGINT,
  creation BIGINT,
  expiration BIGINT,
  tags SET<TEXT>,
  PRIMARY KEY (entity, aggregate, key)
);

CREATE TABLE tag_expiration (
  tag TEXT,
  window TEXT,
  expiration BIGINT,
  aggregate TEXT,
  entity TEXT,
  key TEXT,
  ttl BIGINT,
  creation BIGINT,
  tags SET<TEXT>,
  PRIMARY KEY ((tag, window), expiration, aggregate, entity)
);

CREATE TABLE key_expiration (
  key TEXT,
  window TEXT,
  expiration BIGINT,
  aggregate TEXT,
  entity TEXT,
  ttl BIGINT,
  creation BIGINT,
  tags SET<TEXT>,
  PRIMARY KEY ((key, window), expiration, aggregate, entity)
);

CREATE TABLE write_progress (
  id TEXT,
  sequence_nr BIGINT,
  PRIMARY KEY (id)
);
